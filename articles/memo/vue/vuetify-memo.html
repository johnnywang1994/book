  <v-dialog :fullscreen="$vuetify.breakpoint.mobile">

  </v-dialog>
</template>

import Vue from 'vue'
import Vuetify from 'vuetify/lib'

export default new Vuetify({
  breakpoint: {
    mobileBreakpoint: 'sm' // This is equivalent to a value of 960
  },
})
